import tkinter as tk
import requests
import pprint

base = tk.Tk()
base.title("天気")
city_tuple = ("札幌", "東京", "大阪")
city_dic = {"札幌":"016010", "東京":"130010", "大阪":"270000"}
check_value ={}

for i in city_tuple:
    check_value[i] = tk.BooleanVar()
    tk.Checkbutton(base, variable=check_value[i], text=i).pack(anchor=tk.W)
